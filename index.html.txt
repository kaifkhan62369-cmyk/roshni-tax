    <script>
        const fileInput = document.getElementById('file-input');
        const results = document.getElementById('results');

        fileInput.addEventListener('change', function(e) {
            if(e.target.files.length > 0) {
                let file = e.target.files[0];
                let formData = new FormData();
                formData.append("file", file);

                // PASTE YOUR RENDER URL IN THE LINE BELOW
                fetch("https://roshni-tax-backend.onrender.com/process-report/", {
                    method: "POST",
                    body: formData
                })
                .then(response => response.json())
                .then(data => {
                    results.style.display = 'block';
                    document.getElementById('status-message').innerText = file.name + " processed! Detected: " + data.marketplace_detected;
                })
                .catch(error => {
                    alert("Error connecting to the tax engine. Check your Render URL.");
                });
            }
        });
    </script>
